                            </div>
                          </motion.div>
                        ))}
                      </div>
                    </motion.div>
                  ) : activeTab === 'explore' ? (
                    renderExplore()
                  ) : activeTab === 'saved' ? (
                    renderSaved()
                  ) : activeTab === 'profile' ? (
                    renderProfile()
                  ) : null}
                </AnimatePresence>
              </main>

              <AnimatePresence>
                {selectedPost && (
                  <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} className="fixed inset-0 bg-slate-900/10 backdrop-blur-md z-[110] flex justify-center">
                    <div className="w-full max-w-[480px] h-full relative" onClick={() => setSelectedPost(null)}>
                      <motion.div layoutId={`post-container-${selectedPost.id}`} onClick={(e) => e.stopPropagation()} className="absolute bottom-0 w-full h-[88%] bg-white/70 backdrop-blur-[60px] rounded-t-[40px] p-8 shadow-2xl flex flex-col border-t border-white/60 overflow-y-auto no-scrollbar">
                        <div className="w-10 h-1 bg-slate-200 rounded-full mx-auto mb-8 shrink-0" />
                        <div className="flex items-center justify-between mb-8">
                          <div className="flex items-center gap-4"><img src={selectedPost.logo} alt="" className="w-9 h-9 object-contain p-1.5 bg-white rounded-xl border border-white/60 shadow-sm" /><div><p className="text-sm font-black text-slate-900 uppercase leading-none mb-1">{selectedPost.company}</p><p className="text-[10px] text-slate-400 font-bold">{selectedPost.date}</p></div></div>
                          <button onClick={() => setSelectedPost(null)} className="p-2.5 bg-white/60 backdrop-blur-md rounded-full text-slate-400 border border-white/60 shadow-sm"><X size={18} /></button>
                        </div>
                        {selectedPost.thumbnail && (
                          <div className="w-full h-44 mb-6 shrink-0 rounded-3xl overflow-hidden shadow-md">
                            <img src={selectedPost.thumbnail} alt="" className="w-full h-full object-cover" />
                          </div>
                        )}
                        <h2 className="text-xl font-black text-slate-800 leading-tight mb-5">{selectedPost.title}</h2>
                        <div className="text-slate-600 text-[15px] leading-relaxed mb-24 font-medium">{selectedPost.content || selectedPost.excerpt}</div>
                        <div className="mt-auto absolute bottom-8 left-8 right-8 flex gap-3">
                          <button className="flex-1 py-4 bg-slate-900 text-white rounded-2xl font-black text-xs uppercase shadow-xl flex items-center justify-center gap-2 active:scale-95 transition-transform"><ExternalLink size={18} /> 원문 읽기</button>
                          <button onClick={(e) => toggleSave(e, selectedPost.id)} className={`w-14 h-14 rounded-2xl flex items-center justify-center backdrop-blur-md border ${saved.includes(selectedPost.id) ? 'bg-indigo-50 border-indigo-100 text-indigo-600 shadow-lg' : 'bg-white/60 border-white/80 text-slate-300 shadow-sm'}`}><Bookmark size={22} fill={saved.includes(selectedPost.id) ? "currentColor" : "none"} /></button>
                        </div>
                      </motion.div>
                    </div>
                  </motion.div>
                )}
              </AnimatePresence>

              <nav className="absolute bottom-0 w-full bg-white/30 backdrop-blur-[40px] border-t border-white/20 px-8 pt-3 pb-8 flex justify-between items-center z-[60]">
                <TabButton active={activeTab === 'home'} onClick={() => setActiveTab('home')} icon={<Home size={18} />} label="Home" />
                <TabButton active={activeTab === 'explore'} onClick={() => setActiveTab('explore')} icon={<Compass size={18} />} label="Explore" />
                <TabButton active={activeTab === 'saved'} onClick={() => setActiveTab('saved')} icon={<Bookmark size={18} />} label="Saved" />
                <TabButton active={activeTab === 'profile'} onClick={() => setActiveTab('profile')} icon={<User size={18} />} label="Me" />
              </nav>
            </motion.div>
          )}
        </AnimatePresence>
      </div>
    </div>
  );
};

const TabButton = ({ active, icon, label, onClick }) => (
  <button onClick={onClick} className={`flex flex-col items-center gap-1 transition-all relative ${active ? 'text-slate-900' : 'text-slate-400'}`}>
    <motion.div whileTap={{ y: -3, scale: 1.15 }} className="relative z-10">{icon}</motion.div>
    <span className={`text-[7px] font-black uppercase tracking-widest ${active ? 'text-slate-900' : 'text-slate-400'}`}>
      {label}
    </span>
    {active && <motion.div layoutId="activeTabGlow" className="absolute -top-1 -left-2 -right-2 -bottom-1 bg-white/70 blur-xl rounded-full -z-10" />}
  </button>
);

export default App;
